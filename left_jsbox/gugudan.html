Java Script Code

<hr> <script> document.write('<table class="gugudan" width="400px" style="text-align: center;">'); for (var I = 1; I <= 3; I++) { if (I > 1) { document.write('<tr class="space"><td colspan="3"></td></tr>'); } for (var M = 1; M <= 9; M++) { document.write('<tr>'); for (var N = 1; N <= 3; N++) { var num = 3 * (I - 1) + N; document.write('<td>'); document.write(num + " x " + M + " = " + num * M + "<br>"); document.write('</td>'); } document.write('</tr>'); } } document.write('</table>'); </script>
